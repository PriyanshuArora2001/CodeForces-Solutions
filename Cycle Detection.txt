bool has_cycle(SinglyLinkedListNode* head) 
{
    SinglyLinkedListNode* slow=head;
    SinglyLinkedListNode* fast=head->next;
    while(fast!=nullptr && fast->next!=nullptr)
    {
        if(slow==fast)
            return 1;
        slow=slow->next;
        fast=fast->next->next;
    }
    return 0;
}
